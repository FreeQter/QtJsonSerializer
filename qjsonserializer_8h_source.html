168 template<typename T>
169 bool QJsonSerializer::registerListConverters() {
170  auto ok1 = QMetaType::registerConverter<QList<T>, QVariantList>([](const QList<T> &list) -> QVariantList {
171  QVariantList l;
172  foreach(auto v, list)
173  l.append(QVariant::fromValue(v));
174  return l;
175  });
176 
177  auto ok2 = QMetaType::registerConverter<QVariantList, QList<T>>([](const QVariantList &list) -> QList<T> {
178  QList<T> l;
179  foreach(auto v, list) {
180  auto vt = v.type();
181  if(v.convert(qMetaTypeId<T>()))
182  l.append(v.value<T>());
183  else {
184  qWarning() << "Conversion to"
185  << QMetaType::typeName(qMetaTypeId<QList<T>>())
186  << "failed, could not convert element of type"
187  << QMetaType::typeName(vt);
188  l.append(T());
189  }
190  }
191  return l;
192  });
193 
194  return ok1 && ok2;
195 }

198 bool QJsonSerializer::registerMapConverters()
199 {
200  auto ok1 = QMetaType::registerConverter<QMap<QString, T>, QVariantMap>([](const QMap<QString, T> &map) -> QVariantMap {
201  QVariantMap m;
202  for(auto it = map.constBegin(); it != map.constEnd(); ++it)
203  m.insert(it.key(), QVariant::fromValue(it.value()));
204  return m;
205  });
206 
207  auto ok2 = QMetaType::registerConverter<QVariantMap, QMap<QString, T>>([](const QVariantMap &map) -> QMap<QString, T> {
208  QMap<QString, T> m;
209  for(auto it = map.constBegin(); it != map.constEnd(); ++it) {
210  auto v = it.value();
211  auto vt = v.type();
212  if(v.convert(qMetaTypeId<T>()))
213  m.insert(it.key(), v.value<T>());
214  else {
215  qWarning() << "Conversion to"
216  << QMetaType::typeName(qMetaTypeId<QMap<QString, T>>())
217  << "failed, could not convert element value of type"
218  << QMetaType::typeName(vt);
219  m.insert(it.key(), T());
220  }
221  }
222  return m;
223  });
224 
225  return ok1 && ok2;

237 bool QJsonSerializer::registerPointerConverters()
238 {
239  static_assert(std::is_base_of<QObject, T>::value, "T must inherit QObject");
240  auto ok1 = QMetaType::registerConverter<QSharedPointer<QObject>, QSharedPointer<T>>([](const QSharedPointer<QObject> &object) -> QSharedPointer<T> {
241  return object.objectCast<T>();
242  });
243  auto ok2 = QMetaType::registerConverter<QSharedPointer<T>, QSharedPointer<QObject>>([](const QSharedPointer<T> &object) -> QSharedPointer<QObject> {
244  return object.template staticCast<QObject>();//must work, because of static assert
245  });
246  auto ok3 = QMetaType::registerConverter<QPointer<QObject>, QPointer<T>>([](const QPointer<QObject> &object) -> QPointer<T> {
247  return qobject_cast<T*>(object.data());
248  });
249  auto ok4 = QMetaType::registerConverter<QPointer<T>, QPointer<QObject>>([](const QPointer<T> &object) -> QPointer<QObject> {
250  return static_cast<QObject*>(object.data());
251  });
252  return ok1 && ok2 && ok3 && ok4;
