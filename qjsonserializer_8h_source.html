220  QVariantList l;

227  QList<T> l;
228  for(auto v : list) {
229  auto vt = v.type();
230  if(v.convert(qMetaTypeId<T>()))
231  l.append(v.value<T>());
232  else {
233  qWarning() << "Conversion to"
234  << QMetaType::typeName(qMetaTypeId<QList<T>>())
235  << "failed, could not convert element of type"
236  << QMetaType::typeName(vt);
237  l.append(T());
238  }
239  }
240  return l;
241  });
