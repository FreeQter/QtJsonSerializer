99 bool QJsonSerializer::registerListConverters() {
100  auto ok1 = QMetaType::registerConverter<QList<T>, QVariantList>([](const QList<T> &list) -> QVariantList {
101  QVariantList l;
102  foreach(auto v, list)
103  l.append(QVariant::fromValue(v));
104  return l;
105  });
106 
107  auto ok2 = QMetaType::registerConverter<QVariantList, QList<T>>([](const QVariantList &list) -> QList<T> {
108  QList<T> l;
109  foreach(auto v, list) {
110  auto vt = v.type();
111  if(v.convert(qMetaTypeId<T>()))
112  l.append(v.value<T>());
113  else {
114  qWarning() << QByteArray("Conversion to")
115  << QMetaType::typeName(qMetaTypeId<QList<T>>())
116  << QByteArray("failed, could not convert element of type")
117  << QMetaType::typeName(vt);
118  l.append(T());
119  }
120  }
121  return l;
122  });
123 
124  auto ok3 = QMetaType::registerComparators<QList<T>>();
125 
126  return ok1 && ok2 && ok3;
127 }
