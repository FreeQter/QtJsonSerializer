127 template<typename T>
128 bool QJsonSerializer::registerListConverters() {
129  auto ok1 = QMetaType::registerConverter<QList<T>, QVariantList>([](const QList<T> &list) -> QVariantList {
130  QVariantList l;
131  foreach(auto v, list)
132  l.append(QVariant::fromValue(v));
133  return l;
134  });
135 
136  auto ok2 = QMetaType::registerConverter<QVariantList, QList<T>>([](const QVariantList &list) -> QList<T> {
137  QList<T> l;
138  foreach(auto v, list) {
139  auto vt = v.type();
140  if(v.convert(qMetaTypeId<T>()))
141  l.append(v.value<T>());
142  else {
143  qWarning() << QByteArray("Conversion to")
144  << QMetaType::typeName(qMetaTypeId<QList<T>>())
145  << QByteArray("failed, could not convert element of type")
146  << QMetaType::typeName(vt);
147  l.append(T());
148  }
149  }
150  return l;
151  });
152 
153  return ok1 && ok2;
154 }
