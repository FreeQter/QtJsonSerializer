42 QJsonValue QJsonSerializer::serializeVariant(int propertyType, const QVariant &value) const
43 {
44  auto convertValue = value;
45  if((propertyType == QVariant::List) ||
46  (convertValue.canConvert(QVariant::List) && convertValue.convert(QVariant::List))) {
47  return serializeList(propertyType, value.toList());
48  } else {
49  convertValue = value;
50  auto flags = QMetaType::typeFlags(propertyType);
51 
52  if(flags.testFlag(QMetaType::IsGadget)) {
53  auto metaObject = QMetaType::metaObjectForType(propertyType);
54  return serializeGadget(value.constData(), metaObject);
55  } else if(flags.testFlag(QMetaType::PointerToQObject) ||
56  (convertValue.canConvert(QMetaType::QObjectStar) && convertValue.convert(QMetaType::QObjectStar))) {
57  auto object = convertValue.value<QObject*>();
58  if(object)
59  return serializeObject(object);
60  else
61  return QJsonValue::Null;
62  } else
63  return serializeValue(propertyType, value);
64  }
65 }

112 {

135 QVariant QJsonSerializer::deserializeVariant(int propertyType, const QJsonValue &value, QObject *parent) const
136 {
137  QVariant variant;
138  if(value.isArray()) {
139  variant = deserializeList(propertyType, value.toArray(), parent);
140  } else if(value.isObject() || value.isNull()) {
141  auto flags = QMetaType::typeFlags(propertyType);
142 
143  if(flags.testFlag(QMetaType::IsGadget)) {
144  if(!value.isNull()) {
145  QVariant gadget(propertyType, nullptr);
146  deserializeGadget(value.toObject(), propertyType, gadget.data());
147  variant = gadget;
148  }
149  } else if(flags.testFlag(QMetaType::PointerToQObject)) {
150  if(value.isNull())
151  variant = QVariant::fromValue<QObject*>(nullptr);
152  else {
153  auto object = deserializeObject(value.toObject(), QMetaType::metaObjectForType(propertyType), parent);
154  variant = QVariant::fromValue(object);
155  }
156  } else
157  variant = deserializeValue(propertyType, value);
158  } else
159  variant = deserializeValue(propertyType, value);
160 
161  if(propertyType != QMetaType::UnknownType) {
162  auto vType = variant.typeName();
163  if(variant.canConvert(propertyType) && variant.convert(propertyType))
164  return variant;
165  else if(_allowNull && value.isNull())
166  return QVariant();
167  else {
168  throw QJsonDeserializationException(QByteArray("Failed to convert deserialized variant of type ") +
169  (vType ? vType : "<unknown>") +
170  QByteArray(" to property type ") +
171  QMetaType::typeName(propertyType));
172  }
173  } else
174  return variant;
175 }

177 QObject *QJsonSerializer::deserializeObject(const QJsonObject &jsonObject, const QMetaObject *metaObject, QObject *parent) const
178 {
179  //try to construct the object
180  auto object = metaObject->newInstance(Q_ARG(QObject*, parent));
181  if(!object)
182  throw QJsonDeserializationException(QByteArray("Failed to construct object of type ") +
183  metaObject->className() +
184  QByteArray(" (Does the constructor \"Q_INVOKABLE class(QObject*);\" exist?)"));
185 
186  //now deserialize all json properties
187  for(auto it = jsonObject.constBegin(); it != jsonObject.constEnd(); it++) {
188  auto propIndex = metaObject->indexOfProperty(qUtf8Printable(it.key()));
189  auto type = propIndex != -1 ?
190  metaObject->property(propIndex).userType() :
191  QMetaType::UnknownType;
192  object->setProperty(qUtf8Printable(it.key()), deserializeVariant(type, it.value(), object));
193  }
194 
195  return object;
196 }

198 void QJsonSerializer::deserializeGadget(const QJsonObject &jsonObject, int typeId, void *gadgetPtr) const
199 {
200  auto metaObject = QMetaType::metaObjectForType(typeId);
201  if(!QMetaType::construct(typeId, gadgetPtr, nullptr))
202  throw QJsonDeserializationException(QByteArray("Failed to construct gadget of type ") + QMetaType::typeName(typeId));
203 
204  //now deserialize all json properties
205  for(auto it = jsonObject.constBegin(); it != jsonObject.constEnd(); it++) {
206  auto propIndex = metaObject->indexOfProperty(qUtf8Printable(it.key()));
207  if(propIndex != -1) {
208  auto property = metaObject->property(propIndex);
209  property.writeOnGadget(gadgetPtr, deserializeVariant(property.userType(), it.value(), nullptr));
210  }
211  }
